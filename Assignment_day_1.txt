def remove_punctuation(txt):
    result=""
    punct='''!"#$%&'()*+,-./:;<=>?@[\]^_`{|}~'''
    for ch in txt:
        if ch not in punct:
            result+=ch
    return result

Q2:

"""Count Even and Odd Numbers

Take a list of numbers as input (comma-separated).

Count how many are even and how many are odd.

Print results."""

ans:
numbers=input("Enter comma seperated numbers:")
splitted_numbers=[int(x) for x in numbers.split(",")]
print(splitted_numbers)

count_even=0
count_odd=0
for i in splitted_numbers:
    if i%2==0:
        count_even+=1
    else:
        count_odd+=1
print("Even Numbers:",count_even)
print("odd numbers:",count_odd)
